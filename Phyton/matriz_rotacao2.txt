import numpy as np
#Função que retorna a matriz de rotação em z

def THz(theta, Dx, Dy, Dz):
  return np.array([
      [np.cos(theta), -np.sin(theta),0 , Dx],
      [np.sin(theta), np.cos(theta),0 , Dy],
      [0,               0,           1 , Dz],
      [0,               0,           0 ,  1]
  ])
 
print('Hello world')
theta = np.deg2rad(30)
T = THz(theta, 10, 5, 0)

pb = np.array([3, 7, 0, 1])

pa = T @ pb
print(pa)
